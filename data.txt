    <script>
        const genreSelect = document.getElementById('genreSelect');
        const genreMoviesContainer = document.getElementById('genreMovies');

        genreSelect.addEventListener('change', async (event) => {
            const genreId = event.target.value;
            if (!genreId) return;

            // Show loading spinner
            genreMoviesContainer.innerHTML = '<div class="text-center">Loading...</div>';

            try {
                // Fetch movies for the selected genre
                const response = await fetch(`/genre/${genreId}`);
                const data = await response.text();

                // Parse the HTML response
                const parser = new DOMParser();
                const doc = parser.parseFromString(data, 'text/html');
                const movies = doc.querySelectorAll('.card');

                // Clear the container and append the movies
                genreMoviesContainer.innerHTML = '';
                movies.forEach(movie => {
                    genreMoviesContainer.appendChild(movie);
                });
            } catch (error) {
                console.error('Error fetching genre movies:', error);
                genreMoviesContainer.innerHTML = '<div class="text-center text-danger">Failed to load movies. Please try again.</div>';
            }
        });
    </script>
